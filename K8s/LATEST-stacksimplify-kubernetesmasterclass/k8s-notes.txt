Step-01: Create EKS Cluster using eksctl

eksctl create cluster --name=eksdemo1 \
                      --region=ap-southeast-1 \
                      --zones=ap-southeast-1a,ap-southeast-1b \
                      --without-nodegroup 

eksctl create cluster --name=my-eks \
                      --region=ap-southeast-1 \
                      --zones=ap-southeast-1a,ap-southeast-1b \
                      --nodegroup-name=my-eks-ng \
                      --node-type=t2.micro \
                      --nodes=2 \
                      --nodes-min=2 \
                      --nodes-max=4 \
                      --node-volume-size=20 \
                      --ssh-access \
                      --ssh-public-key=encina-ta-key \
                      --managed \
                      --asg-access \
                      --external-dns-access \
                      --full-ecr-access \
                      --appmesh-access \
                      --alb-ingress-access \
                      --vpc-cidr=encina-test-vpc \
                      --vpc-private-subnets=encina-test-pri-sub-a,encina-test-pri-sub-b		      
                      

# Get List of clusters
eksctl get cluster 

# Delete cluster
eksctl delete cluster --name=my-eks --region=ap-southeast-1


Step-02: Create & Associate IAM OIDC Provider for our EKS Cluster

eksctl utils associate-iam-oidc-provider \
    --region ap-southeast-1 \
    --cluster my-eks \
    --approve

Step-04: Create Node Group with additional Add-Ons in Public Subnets

# Create Public Node Group   
eksctl create nodegroup --cluster=eksdemo1 \
                       --region=ap-southeast-1 \
                       --name=eksdemo1-ng-public1 \
                       --node-type=t2.micro \
                       --nodes=2 \
                       --nodes-min=2 \
                       --nodes-max=4 \
                       --node-volume-size=20 \
                       --ssh-access \
                       --ssh-public-key=encina-ta-key \
		       --subnet-ids=
                       --managed \
                       --asg-access \
                       --external-dns-access \
                       --full-ecr-access \
                       --appmesh-access \
